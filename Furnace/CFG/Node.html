def inspect
  if @label && @instructions
    "<#{@label}:#{@instructions.join ", "}>"
  elsif @label
    "<#{@label}>"
  elsif @instructions
    "<!unlabeled:#{@instructions.join ", "}>"
  else
    "<!exit#{metadata}>"
  end
end
